(* This module implements the K-Nearest Neighbor algorithm.  *)

IMPORT KDTree;

CONST
    (* Maximum number of neighbors to search for. *)
    MaxK = 20;  

TYPE
    (* Represents the Neighbors with their distances.  *)
    Neighbor = RECORD
        point: KDTree.Point;
        dist: REAL
    END;

    NeighborList = ARRAY MaxK OF Neighbor;
